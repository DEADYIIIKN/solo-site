```typescript
import { describe, it, expect, beforeEach } from "vitest";
import { checkRateLimit, __resetRateLimit } from "@/lib/leads/rate-limit";

describe("checkRateLimit", () => {
  beforeEach(() => __resetRateLimit());

  it("allows first 10 requests from same IP", () => {
    const now = 1_000_000;
    for (let i = 0; i < 10; i++) {
      expect(checkRateLimit("1.1.1.1", now + i).allowed).toBe(true);
    }
  });

  it("blocks 11th request within 5min window", () => {
    const now = 1_000_000;
    for (let i = 0; i < 10; i++) checkRateLimit("1.1.1.1", now + i);
    const res = checkRateLimit("1.1.1.1", now + 10);
    expect(res.allowed).toBe(false);
    expect(res.retryAfterMs).toBeGreaterThan(0);
    expect(res.retryAfterMs).toBeLessThanOrEqual(5 * 60 * 1000);
  });

  it("treats different IPs independently", () => {
    const now = 1_000_000;
    for (let i = 0; i < 10; i++) checkRateLimit("1.1.1.1", now + i);
    for (let i = 0; i < 10; i++) {
      expect(checkRateLimit("2.2.2.2", now + i).allowed).toBe(true);
    }
  });

  it("slides window — old requests expire after 5min", () => {
    const start = 1_000_000;
    for (let i = 0; i < 10; i++) checkRateLimit("1.1.1.1", start + i);
    // Через 5 мин + 1мс старые timestamps вне окна
    const later = start + 5 * 60 * 1000 + 1;
    expect(checkRateLimit("1.1.1.1", later).allowed).toBe(true);
  });

  it("__resetRateLimit clears storage", () => {
    const now = 1_000_000;
    for (let i = 0; i < 10; i++) checkRateLimit("1.1.1.1", now + i);
    __resetRateLimit();
    expect(checkRateLimit("1.1.1.1", now + 100).allowed).toBe(true);
  });
});
```

```typescript
import { describe, it, expect } from "vitest";
import { leadSchema } from "@/lib/leads/validation";

const valid = {
  name: "Иван",
  phone: "+7 (900) 123-45-67",
  message: "тестовое сообщение",
  consent: true,
  contactMethod: "call" as const,
  source: "hero-cta",
};

describe("leadSchema", () => {
  it("accepts valid payload", () => {
    expect(leadSchema.safeParse(valid).success).toBe(true);
  });

  it("rejects empty name", () => {
    const r = leadSchema.safeParse({ ...valid, name: "" });
    expect(r.success).toBe(false);
  });

  it("rejects invalid phone format", () => {
    const r = leadSchema.safeParse({ ...valid, phone: "abc" });
    expect(r.success).toBe(false);
  });

  it("rejects consent=false", () => {
    const r = leadSchema.safeParse({ ...valid, consent: false });
    expect(r.success).toBe(false);
  });

  it("rejects unknown contactMethod", () => {
    const r = leadSchema.safeParse({ ...valid, contactMethod: "email" });
    expect(r.success).toBe(false);
  });

  it("rejects empty source", () => {
    const r = leadSchema.safeParse({ ...valid, source: "" });
    expect(r.success).toBe(false);
  });

  it("accepts missing message (optional)", () => {
    const { message, ...rest } = valid;
    void message;
    const r = leadSchema.safeParse(rest);
    expect(r.success).toBe(true);
  });
});
```
