    Создать `src/widgets/tg-popup/ui/tg-popup-host.tsx` (`"use client"`):

    ```typescript
    "use client";
    import { useCallback, useEffect, useState } from "react";

    import { useViewportLayout } from "@/shared/lib/use-viewport-layout";
    import { useActivityTimer } from "@/shared/lib/use-activity-timer";

    import { TgPopup } from "./tg-popup";

    const STORAGE_KEY = "tg-popup-dismissed";
    const TRIGGER_MS = 60_000;

    export function TgPopupHost() {
      const ctaHref = process.env.NEXT_PUBLIC_TG_CHANNEL_URL ?? "";
      const variant = useViewportLayout();

      // sessionStorage gate — initial check во время mount.
      const [dismissed, setDismissed] = useState<boolean | null>(null);
      useEffect(() => {
        if (typeof window === "undefined") return;
        try {
          setDismissed(window.sessionStorage.getItem(STORAGE_KEY) === "1");
        } catch {
          setDismissed(false);
        }
      }, []);

      const [open, setOpen] = useState(false);
      const shouldArmTimer = !!ctaHref && dismissed === false;

      // Timer всегда вызывается (Rules of Hooks), но условный durationMs:
      // если shouldArmTimer === false — передаём Infinity → callback не сработает.
      useActivityTimer(shouldArmTimer ? TRIGGER_MS : Number.POSITIVE_INFINITY, useCallback(() => {
        setOpen(true);
      }, []));

      const handleDismiss = useCallback(() => {
        setOpen(false);
        try {
          window.sessionStorage.setItem(STORAGE_KEY, "1");
        } catch {
          /* noop — приватный режим Safari etc. */
        }
      }, []);

      // Silent skip: env пустой ИЛИ уже dismiss-нуто ИЛИ variant ещё не resolved.
      if (!ctaHref) return null;
      if (dismissed !== false) return null;
      if (!variant) return null;

      return (
        <TgPopup
          variant={variant}
          open={open}
          onDismiss={handleDismiss}
          ctaHref={ctaHref}
        />
      );
    }
    ```

    **Важно:** `useActivityTimer` в 10-01 должен корректно обработать `Number.POSITIVE_INFINITY` (accumulator никогда не дойдёт). Если хук этого не поддерживает — добавить early-return в хуке для `!Number.isFinite(durationMs)`. (Скорее всего достаточно естественной семантики `accumulatedMs >= durationMs` — `>= Infinity` всегда false.)

    Альтернатива (если хочется чище): сделать TgPopupHost условным wrapper'ом — внутри сделать inner-компонент `<TgPopupHostActive>` который безусловно вызывает useActivityTimer, и рендерить его только когда shouldArmTimer === true. Это **предпочтительный путь** — лучше явный gate чем Infinity-magic:

    ```typescript
    export function TgPopupHost() {
      const ctaHref = process.env.NEXT_PUBLIC_TG_CHANNEL_URL ?? "";
      const variant = useViewportLayout();
      const [dismissed, setDismissed] = useState<boolean | null>(null);
      useEffect(() => {
        if (typeof window === "undefined") return;
        try {
          setDismissed(window.sessionStorage.getItem(STORAGE_KEY) === "1");
        } catch { setDismissed(false); }
      }, []);

      if (!ctaHref) return null;
      if (dismissed !== false) return null;
      if (!variant) return null;

      return <TgPopupHostActive variant={variant} ctaHref={ctaHref} />;
    }

    function TgPopupHostActive({ variant, ctaHref }: { variant: TgPopupVariant; ctaHref: string }) {
      const [open, setOpen] = useState(false);
      useActivityTimer(TRIGGER_MS, useCallback(() => setOpen(true), []));
      const handleDismiss = useCallback(() => {
        setOpen(false);
        try { window.sessionStorage.setItem(STORAGE_KEY, "1"); } catch {}
      }, []);
      return <TgPopup variant={variant} open={open} onDismiss={handleDismiss} ctaHref={ctaHref} />;
    }
    ```
