    ```typescript
    import { expect, test } from "./_fixtures";

    const POPUP_ROOT = '[data-testid="tg-popup-root"]';
    const POPUP_CLOSE = '[data-testid="tg-popup-close"]';
    const POPUP_CTA = '[data-testid="tg-popup-cta"]';
    const TRIGGER_MS = 60_000;

    test.describe("TG popup", () => {
      test.beforeEach(async ({ page }) => {
        await page.clock.install({ time: new Date("2026-04-27T10:00:00Z") });
      });

      test("appears after 60s of activity", async ({ page }) => {
        await page.goto("/");
        // Эмулируем активность: mousemove каждые 5 виртуальных секунд.
        await page.mouse.move(100, 100);
        for (let i = 0; i < 12; i++) {
          await page.clock.fastForward(5_000);
          await page.mouse.move(100 + i, 100 + i);
        }
        await expect(page.locator(POPUP_ROOT)).toBeVisible({ timeout: 5_000 });
      });

      test("dismiss via ✕ persists in sessionStorage", async ({ page }) => {
        await page.goto("/");
        await page.mouse.move(100, 100);
        await page.clock.fastForward(TRIGGER_MS);
        await page.mouse.move(101, 101);
        await expect(page.locator(POPUP_ROOT)).toBeVisible();
        await page.locator(POPUP_CLOSE).click();
        await expect(page.locator(POPUP_ROOT)).toBeHidden();
        const dismissed = await page.evaluate(() =>
          window.sessionStorage.getItem("tg-popup-dismissed"),
        );
        expect(dismissed).toBe("1");
      });

      test("dismiss via overlay click", async ({ page }) => {
        await page.goto("/");
        await page.mouse.move(100, 100);
        await page.clock.fastForward(TRIGGER_MS);
        await page.mouse.move(101, 101);
        await expect(page.locator(POPUP_ROOT)).toBeVisible();
        // Клик в верхний-левый угол (overlay backdrop area за пределами card).
        await page.mouse.click(5, 5);
        await expect(page.locator(POPUP_ROOT)).toBeHidden();
      });

      test("dismiss via ESC", async ({ page }) => {
        await page.goto("/");
        await page.mouse.move(100, 100);
        await page.clock.fastForward(TRIGGER_MS);
        await page.mouse.move(101, 101);
        await expect(page.locator(POPUP_ROOT)).toBeVisible();
        await page.keyboard.press("Escape");
        await expect(page.locator(POPUP_ROOT)).toBeHidden();
      });

      test("does not reappear after reload (sessionStorage gate)", async ({ page }) => {
        await page.goto("/");
        await page.mouse.move(100, 100);
        await page.clock.fastForward(TRIGGER_MS);
        await page.mouse.move(101, 101);
        await page.locator(POPUP_CLOSE).click();
        await page.reload();
        await page.clock.fastForward(TRIGGER_MS + 5_000);
        await page.mouse.move(102, 102);
        await expect(page.locator(POPUP_ROOT)).toBeHidden();
      });

      test("reappears in fresh browser context", async ({ browser }) => {
        const ctx1 = await browser.newContext();
        const page1 = await ctx1.newPage();
        await page1.clock.install({ time: new Date("2026-04-27T10:00:00Z") });
        await page1.goto("/");
        await page1.mouse.move(100, 100);
        await page1.clock.fastForward(TRIGGER_MS);
        await page1.mouse.move(101, 101);
        await page1.locator(POPUP_CLOSE).click();
        await ctx1.close();

        const ctx2 = await browser.newContext();
        const page2 = await ctx2.newPage();
        await page2.clock.install({ time: new Date("2026-04-27T10:00:00Z") });
        await page2.goto("/");
        await page2.mouse.move(100, 100);
        await page2.clock.fastForward(TRIGGER_MS);
        await page2.mouse.move(101, 101);
        await expect(page2.locator(POPUP_ROOT)).toBeVisible();
        await ctx2.close();
      });

      test("CTA opens TG URL with target=_blank + rel=noopener", async ({ page }) => {
        await page.goto("/");
        await page.mouse.move(100, 100);
        await page.clock.fastForward(TRIGGER_MS);
        await page.mouse.move(101, 101);
        const cta = page.locator(POPUP_CTA);
        await expect(cta).toBeVisible();
        await expect(cta).toHaveAttribute("target", "_blank");
        await expect(cta).toHaveAttribute("rel", /noopener/);
        const href = await cta.getAttribute("href");
        expect(href).toMatch(/^https?:\/\/t\.me\//);
      });
    });
    ```

    - Для visibility-теста (опционально) не покрываем в E2E — слишком хрупко на CI; покрыто unit'ами в 10-01.
